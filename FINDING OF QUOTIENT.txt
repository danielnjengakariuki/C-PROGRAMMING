//ARITHMETIC OPERATORS- FINDING THE QUOTIENT OF NUMBERS
#include <stdio.h>

int main() 
{
    int x, y, quotient;
    printf("enter the value of x \n");
    scanf("%d", &x);
    printf("enter the value of y \n");
    scanf("%d", &y);
    quotient = x/y;
    printf("quotient = %d", quotient);
    return 0;
}
